src/
├── prisma/                 
│   ├── prisma.service.ts   // Connects to your SQLite database
│   └── prisma.module.ts    // Makes the DB available to the rest of the app
├── auth/                   
│   ├── auth.controller.ts  // Handles /signup and /login routes
│   ├── auth.service.ts     // Verifies passwords and creates JWT tokens
│   └── auth.module.ts      
├── users/                  
│   ├── users.controller.ts // Handles Admin APIs like getting all users
│   ├── users.service.ts    // Logic for finding or updating user roles
│   └── users.module.ts     
├── blogs/                  
│   ├── blogs.controller.ts // Handles CRUD routes for blogs
│   ├── blogs.service.ts    // Logic for saving or deleting blogs in Prisma
│   └── blogs.module.ts     
├── app.module.ts           // The master file that imports Auth, Users, and Blogs
└── main.ts                 // The entry point that starts your server
